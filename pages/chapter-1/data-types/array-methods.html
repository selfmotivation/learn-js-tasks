Переведите текст вида border-left-width в borderLeftWidth

Напишите функцию camelize(str), которая преобразует строки вида «my-short-string» в «myShortString».

То есть дефисы удаляются, а все слова после них получают заглавную букву.

Примеры:

  1. camelize("background-color") == 'backgroundColor';
  2. camelize("list-style-image") == 'listStyleImage';
  3. camelize("-webkit-transition") == 'WebkitTransition';

P.S. Подсказка: используйте split, чтобы разбить строку на массив символов, потом переделайте всё как нужно и методом join соедините обратно.

Решение:

  function camelize(str) {
    return str.split('-').map((el, i) => i === 0 ? el : el[0].toUpperCase() + el.slice(1)).join('');
  }
Фильтрация по диапазону

Напишите функцию filterRange(arr, a, b), которая принимает массив arr, ищет в нём элементы между a и b и отдаёт массив этих элементов.

Функция должна возвращать новый массив и не изменять исходный.

Например:

  1. let arr = [5, 3, 8, 1];
  3. let filtered = filterRange(arr, 1, 4);
  5. alert( filtered ); // 3,1 (совпадающие значения)
  7. alert( arr ); // 5,3,8,1 (без изменений)

Решение:

  function filterRange(arr, a, b) {
    return arr.filter(el => a <= el && el <= b);
  }
Фильтрация по диапазону "на месте"

Напишите функцию filterRangeInPlace(arr, a, b), которая принимает массив arr и удаляет из него все значения кроме тех, которые находятся между a и b. То есть, проверка имеет вид a ≤ arr[i] ≤ b.

Функция должна изменять принимаемый массив и ничего не возвращать.

Например:

  1. let arr = [5, 3, 8, 1];
  3. filterRangeInPlace(arr, 1, 4); // удалены числа вне диапазона 1..4
  5. alert( arr ); // [3, 1]

Решение:

  function filterRangeInPlace(arr, a, b) {
    for (let i = 0; i < arr.length; i++) {
      const el = arr[i];
  
      if (el < a || el > b) {
        arr.splice(i, 1);
        i--;
      }
    }
  }

Скопировать и отсортировать массив

  3. let sorted = copySorted(arr);

Решение:

  function copySorted(arr) {

Создать расширяемый калькулятор

Создайте функцию конструктор Calculator, которая создаёт «расширяемые» объекты калькулятора.

Задание состоит из двух частей.

1.

Во-первых, реализуйте метод calculate(str), который принимает строку типа "1 + 2" в формате «ЧИСЛО оператор ЧИСЛО» (разделено пробелами) и возвращает результат. Метод должен понимать плюс + и минус -.

Пример использования:

  1. let calc = new Calculator;
  3. alert( calc.calculate("3 + 7") ); // 10

2.

Затем добавьте метод addMethod(name, func), который добавляет в калькулятор новые операции. Он принимает оператор name и функцию с двумя аргументами func(a,b), которая описывает его.

Например, давайте добавим умножение *, деление / и возведение в степень **:

  1. let powerCalc = new Calculator;
  2. powerCalc.addMethod("*", (a, b) => a * b);
  3. powerCalc.addMethod("/", (a, b) => a / b);
  4. powerCalc.addMethod("**", (a, b) => a ** b);
  6. let result = powerCalc.calculate("2 ** 3");
  7. alert( result ); // 8

  • Для этой задачи не нужны скобки или сложные выражения.
  • Числа и оператор разделены ровно одним пробелом.
  • Не лишним будет добавить обработку ошибок.

Решение:

  function Calculator() {

    this.methods = {
      "-": (a, b) => a - b,
      "+": (a, b) => a + b
    };
  
    this.calculate = function(str) {
  
      let split = str.split(' '),
        a = +split[0],
        op = split[1],
        b = +split[2]
  
      if (!this.methods[op] || isNaN(a) || isNaN(b)) {
        return NaN;
      }
  
      return this.methods[op](a, b);
    }
  
    this.addMethod = function(name, func) {
      this.methods[name] = func;
    };
  }

Отсортировать пользователей по возрасту

Напишите функцию sortByAge(users), которая принимает массив объектов со свойством age и сортирует их по нему.

Например:

  1. let vasya = { name: "Вася", age: 25 };
  2. let petya = { name: "Петя", age: 30 };
  3. let masha = { name: "Маша", age: 28 };
  5. let arr = [ vasya, petya, masha ];
  7. sortByAge(arr);
  9. // теперь: [vasya, masha, petya]
  10. alert(arr[0].name); // Вася
  11. alert(arr[1].name); // Маша
  12. alert(arr[2].name); // Петя

Решение:

  function sortByAge(arr) {
    arr.sort((a, b) => a.age > b.age ? 1 : -1);
  }
Перемешайте массив

Напишите функцию shuffle(array), которая перемешивает (переупорядочивает случайным образом) элементы массива.

Многократные прогоны через shuffle могут привести к разным последовательностям элементов. Например:

  1. let arr = [1, 2, 3];
  3. shuffle(arr);
  4. // arr = [3, 2, 1]
  6. shuffle(arr);
  7. // arr = [2, 1, 3]
  9. shuffle(arr);
  10. // arr = [3, 1, 2]
  11. // ...

Все последовательности элементов должны иметь одинаковую вероятность. Например, [1,2,3] может быть переупорядочено как [1,2,3] или [1,3,2], или [3,1,2] и т.д., с равной вероятностью каждого случая.

Решение:

  function shuffle(array) {
    for (let i = array.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));

      [array[i], array[j]] = [array[j], array[i]];
    }
  }
Получить средний возраст

Напишите функцию getAverageAge(users), которая принимает массив объектов со свойством age и возвращает средний возраст.

Формула вычисления среднего арифметического значения: (age1 + age2 + ... + ageN) / N.

Например:

  1. let vasya = { name: "Вася", age: 25 };;
  2. let petya = { name: "Петя", age: 30 };
  3. let masha = { name: "Маша", age: 29 };
  5. let arr = [ vasya, petya, masha ];
  7. alert( getAverageAge(arr) ); // (25 + 30 + 29) / 3 = 28

Решение:

  function getAverageAge(users) {
    return users.reduce((prev, user) => prev + user.age, 0) / users.length;
  }

Пусть arr – массив строк.

Напишите функцию unique(arr), которая возвращает массив, содержащий только уникальные элементы arr.

Например:

  1. function unique(arr) {
  2.   /* ваш код */
  3. }
  5. let strings = ["кришна", "кришна", "харе", "харе",
  6.   "харе", "харе", "кришна", "кришна", ":-O"
  7. ];
  9. alert( unique(strings) ); // кришна, харе, :-O

Решение:

  function unique(arr) {
    const resArr = [];
  
    for (let str of arr) {
      if (!resArr.includes(str)) {
        resArr.push(str);
      }
    }
    return resArr;
  }
